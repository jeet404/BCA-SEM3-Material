declare
    grade char(1):='A';
begin
  case grade
    when 'A' then 
        dbms_output.put_line('Excellent');
    when 'B' then
        dbms_output.put_line('Very Good');
    when 'C' then
        dbms_output.put_line('Well Done');
    when 'D' then
        dbms_output.put_line('Passed');
    when 'E' then
        dbms_output.put_line('Better Try Again');  
    else 
        dbms_output.put_line('No Such Grade');
  end case;
end;
